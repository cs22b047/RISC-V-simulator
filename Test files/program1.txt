#bubble sort

.data
    array: .word 1 6 3 9 10
    size: .word 4
.text
    la x5 array
    lw x6 size
    li x7 0
    li x4 0
    loop1:
        add x4 x7 x0
        sub x8 x6 x7
        loop2:
            slli x11 x4 2
            add x11 x11 x5
            lw x10 0(x11)
            lw x9 4(x11)
            blt x10 x9 swap
            addi x4 x4 1
            blt x4 x8 loop2
        addi x7 x7 1
        blt x7 x6 loop1
    li x4 0
    addi x6 x6 1
    j print
    swap:
        sw x10 4(x11)
        sw x9 0(x11)
        j loop1    
    print:
        slli x8 x4 2
        add x8 x8 x5
        lw x10 0(x8)
        li x17 1
        ecall
        addi x4 x4 1
        blt x4 x6 print
